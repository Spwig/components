{# DMARC Configuration #}

{% trans "DMARC (Domain-based Message Authentication)" %}

{% trans "DMARC tells receiving servers what to do if SPF or DKIM checks fail, and where to send reports about email authentication." %}

{% trans "Recommended DMARC Record:" %}

{% trans "Record Type:" %} TXT | {% trans "Hostname:" %} _dmarc.{{ domain }}
v=DMARC1; p=quarantine; rua=mailto:dmarc@{{ domain }}; pct=100; adkim=r; aspf=r

{% trans "DMARC Policy Options:" %}

{% trans "Policy" %} {% trans "Description" %} {% trans "Recommended For" %}
p=none {% trans "Monitor only - no action taken on failures" %} {% trans "Initial testing phase" %}
p=quarantine {% trans "Failed emails go to spam folder" %} {% trans "Production (Recommended)" %}
p=reject {% trans "Failed emails are rejected entirely" %} {% trans "High security requirements" %}
{% trans "Tip:" %} {% trans "If this is your first time setting up DMARC, start with" %} p=none {% trans "to monitor results for a few weeks, then upgrade to" %} p=quarantine.
